    environment:

      PORT: 3000
      DB_HOST: mysql
      DB_PORT: 3306

      DB_USER: item_user

    depends_on:
      - app
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost/health"]
